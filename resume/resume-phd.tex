%----------RESEARCH PROJECTS----------
\section{Research Projects}

\projectHeadingPaper{MelanomaNet: Explainable Deep Learning for Multi-Class Skin Lesion Classification}
{https://github.com/suxrobgm/explainable-melanoma}
{https://arxiv.org/abs/2512.09289}
\begin{itemize}
    \item Developed an explainable deep learning system for multi-class skin lesion classification across all 9 ISIC 2019 diagnostic categories using EfficientNet V2 with high-resolution (384$\times$384) input.
    \item Implemented GradCAM++ attention visualization and novel ABCDE criterion analysis with automated feature extraction (asymmetry quantification, border irregularity, color variation via K-means, diameter calculation).
    \item Designed GradCAM-ABCDE alignment metrics to validate model attention correlation with clinical dermatological features, producing comprehensive interpretability reports.
    \item \textbf{Stack:} Python, PyTorch, EfficientNet V2, GradCAM++, OpenCV, Focal Loss, ISIC 2019 Dataset
\end{itemize}

\projectHeading{LightDepth: Lightweight Monocular Depth Estimation}
{https://github.com/suxrobgm/lightdepth}
\begin{itemize}
    \item Designed a lightweight depth estimation model using ResNet18 encoder with U-Net decoder and skip connections, achieving competitive accuracy with 42\% fewer parameters (14.3M vs 24.8M) than Depth Anything V2.
    \item Achieved 72\% faster inference while outperforming on relative error metrics (5.3\% better AbsRel, 31.1\% better SqRel) on NYU Depth V2.
    \item \textbf{Stack:} Python, PyTorch, ResNet18, U-Net Architecture, L1 Loss, NYU Depth V2 Dataset
\end{itemize}

\projectHeading{FSRCNN: Accelerating Super-Resolution Convolutional Neural Network}
{https://github.com/suxrobgm/fsrcnn}
\begin{itemize}
    \item Implemented FSRCNN for single-image super-resolution (Dong et al., ECCV 2016), achieving 40$\times$ speedup over SRCNN with end-to-end learned upsampling at 2$\times$, 3$\times$, and 4$\times$ scales.
    \item Evaluated on Set5 (+1.78\,dB PSNR, +0.0507 SSIM) and Set14 (+1.26\,dB PSNR, +0.0477 SSIM) with ablation studies.
    \item \textbf{Stack:} Python, PyTorch, Mixed Precision Training (AMP), T91/Set5/Set14/DIV2K Datasets
\end{itemize}

\projectHeading{Bookshelf Scanner: Multi-Modal Book Detection and Recognition}
{https://github.com/suxrobgm/bookshelf-scanner}
\begin{itemize}
    \item Built an end-to-end pipeline combining YOLO segmentation for book instance detection with Moondream2 vision-language model for title/author extraction from bookshelf images.
    \item Integrated object segmentation with LLMs for multi-modal understanding and text recognition.
    \item \textbf{Stack:} Python, YOLO, Moondream2 LLM, Llama CPP, PyTorch, FastAPI, Angular
\end{itemize}

\projectHeading{AoE4 Matchmaking: ML-Assisted Player Matching System}
{https://github.com/suxrobgm/aoe4-matchmaking}
\begin{itemize}
    \item Designed ML-assisted matchmaking combining unsupervised player clustering, XGBoost outcome prediction, and ELO rating for balanced competitive matches.
    \item Applied feature engineering on player statistics and ensemble methods to optimize match pairing.
    \item \textbf{Stack:} Python, XGBoost, K-means Clustering, Pandas, NumPy, FastAPI, Angular
\end{itemize}
